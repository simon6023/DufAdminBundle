<?php

namespace Duf\Bundle\AdminBundle\Annotations;

/**
 * Annotation for indexable entities in DufAdminBundle
 *
 * @author Simon Duflos <simon.duflos@gmail.com>
 *
 * @Annotation
 * @Target("PROPERTY")
 */
final class EditableAnnotation
{

    /**
     * Parameter is_editable
     *
     * @var boolean
     */
    public $is_editable;

    /**
     * Parameter required
     *
     * @var boolean
     */
    public $required;

   /**
     * Parameter label
     *
     * @var string
     */
    public $label;

   /**
     * Parameter type
     *
     * @var string
     */
    public $type;

   /**
     * Parameter filetype
     *
     * @var string
     */
    public $filetype;

    /**
     * Parameter order
     *
     * @var integer
     */
    public $order;

   /**
     * Parameter placeholder
     *
     * @var string
     */
    public $placeholder;

    /**
     * Parameter relation_index
     *
     * @var string
     */
    public $relation_index;

    /**
     * Parameter number_type
     *
     * @var string
     */
    public $number_type;
}
